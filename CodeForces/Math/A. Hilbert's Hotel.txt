void run_case() {
    int n;
    read(n);
    vt<int> a(n);
    read(a);

    set<int> st;
    for(int i = 0; i < n; ++i) {
        int x = (((i + a[i]) % n) + n) % n;
        if(st.find(x) != st.end()) {
            finish("NO");
        }
        st.insert(x);
    }
    print("YES");
}

int main() {
    ios_base::sync_with_stdio(0);

    read(t);
    for(int tc = 1; tc <= t; ++tc) {
        /* write("Case #", tc, run_case); */
        run_case();
    }
