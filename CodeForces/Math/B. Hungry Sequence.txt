#include <bits/stdc++.h>
using namespace std;

// function check whether a number
// is prime or not
bool isPrime(int n)
{
    // Corner cases
    if (n <= 1)
        return false;
    if (n <= 3)
        return true;

    // This is checked so that we can skip
    // middle five numbers in below loop
    if (n % 2 == 0 || n % 3 == 0)
        return false;

    for (int i = 5; i * i <= n; i = i + 6)
        if (n % i == 0 || n % (i + 2) == 0)
            return false;

    return true;
}

// Function to print primes
void printPrime(int n)
{
    int x = 0 ;
    for (int i = 2; i <= pow(10,7); i++)
        {
        if (isPrime(i))
        {
            cout << i << " ";
            x++;
            if(x==n)
                break;
        }
        }
}
// Driver Code
int main()
{
    int n ;
    cin >> n ;
    printPrime(n);
}
