#include<bits/stdc++.h>
#define ll long long
ll gcd( ll a , ll b )
{
     while(b!=0)
     {
         ll a2 = a ;
         a = b ;
         b = a2%b ;
     }
     return a ;
}
ll lcm (ll a , ll b )
{
 return a/gcd(a,b)*b;
}
