#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
ll t ;
cin >> t ;
while(t--)
{
      ll n , flag = 0 ;
      cin >> n ;
      ll arr[n],freqarr[n+1]={0};
      for(int i = 0 ; i < n ; i ++ )
         cin >> arr[i];
         if(n==1)
            cout << 0 << endl;
      for( int i = n-1 ; i >=0 ; i-- )
       {
                if(freqarr[arr[i]]==0)
                {
                  freqarr[arr[i]]++;
                }
                else
                {
                  flag = 1 ;
                  cout << i+1 << endl;
                  break;
                }
       }
        if(flag==0&&n>1)
            cout << 0 << endl;
    }
}
