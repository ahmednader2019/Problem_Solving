#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll n,m,sum=0;
	cin >> n >> m;
	if(n>=m)
        cout << n - m ;
    else
    {
        while(m>n)
        {
           if(m%2==0)
           {
            sum++;
            m/=2;
           }
           else
           {
            m++;
            sum++;
           }
        }
        sum+=n-m;
        cout<< sum << endl;
    }
}
