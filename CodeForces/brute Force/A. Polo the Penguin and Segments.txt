set<pair<int,int>> ss;
int main()
{
    int n,k;cin>>n>>k;
    int counter=0;
    while(n--){
        int x,y;cin>>x>>y;
        ss.insert(make_pair(x,y));
    }
    for (auto const &z : ss) {
        counter+=(z.second-z.first+1);
    }
    if(counter%k==0){
        cout<<0;
    }
    else{
        int c=counter;
        while(counter%k!=0){
            counter+=1;
        }
        cout<<counter-c;
    }
    return 0;
}
